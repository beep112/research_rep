47def print_block_size(block_size, train_data):
48 """
49 This function will be used to print the block size for the data.
50 Another way to describe block size is the amount of context that
51 we will look at for each pass for our model. So in this case the
52 block_size could be variable but for this we will fix it to 8 for
53 this practice model
54
55 @param block_size: the size of context we want to take in
56 @type block_size: int
57 @param train_data: the training data we are using
58 @type train_data: numpy.ndarray
59 @return: void there is nothing returned
60 """
61 print(train_data[: block_size + 1])
62 """
63 x will be used at training data and compared against y which is the
64 actual next token
65 """
66 x = train_data[:block_size]
67 y = train_data[1 : block_size + 1]
68 for t in range(block_size):
69 context = x[: t + 1]
70 target = y[t]
71 print(f"when input is {context} the target: {target}")
