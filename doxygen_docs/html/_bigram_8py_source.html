63 def __init__(self, vocab_size):
64 """
65 @brief constructor for BigramLanguageModel that calls super, and then
66 creates a new embedding table of vocab_size by vocab_size
67
68 @param self: the current BigramLanguageModel object
69 @param vocab_size: the length of the vocab_size
70 @return new BigramLanguageModel
71 """
72 super().__init__()
73 # each token directly reads off the logits for hte next token from a tookup table
74 self.token_embedding_table = nn.Embedding(vocab_size, vocab_size)

76 def forward(self, idx, targets):
77 """
78 @brief goes "ahead" by getting our next set of logits and then returns
79 the logits and loss
80
81 @param self: the current BigramLanguageModel object
82 @param idx: the current context batch
83 @param targets: the tokens that are actually next
84 @return logits, loss: Tuple(logits, loss), which are the logits, and
85 the loss under cross_entropy
86 """
87
88 # idx and targets are both (B, T) tensor of integers
89 logits = self.token_embedding_table(idx) # (B, T, C)
90
91 # negative log likelyhood loss (cross-entropy) how close are logits
92 # to target
93
94 B, T, C = logits.shape
95 logits = logits.view(B * T, C)
96 targets = targets.view(B * T)
97 loss = F.cross_entropy(logits, targets)
98
99 return logits, loss

101 def generate(self, idx, max_new_tokens):
102 """
103 @brief this is the generator of the bigram model, where the model creates tokens
104
105 Idx is the current context of characters in the current batch. And we
106 expent all the B dimension in the T dimension. What is predicted along
107 the T dimension.
108 We take the current idx, focus on the last element in the T dimension.
109 These are then converted to probabilities using softmax, and the sampled
110 using multinomial. We then take the sampled integers, and concat them
111
112 @param self: the current BigramLanguageModel object
113 @param idx: the current context batch (which is dim(B by T))
114 @param max_new_tokens: the amount of new tokens we want to generate
115 @return idx: modified idx with the new generated tokens appended
116 """
117 # idx is (B, T) array of incixes in the current context
118 for _ in range(max_new_tokens):
119 # get the predictions
120 logits, loss = self(idx)
121 # focus only on the last time step
122 logits = logits[:, -1, :] # now dim(B by C)
123 # apply the softmax to get probabilities
124 probs = F.softmax(logits, dim=-1) # dim(B by C)
125 # sample from the distribution
126 idx_next = torch.mulinomial(probs, num_samples=1) # (B, 1)
127 # append sampled index to the running sequence
128 idx = torch.cat((idx, idx_next), dim=1) # dim(B by T-1)
129 return idx
