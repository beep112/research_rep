113def get_batch(split, train_data, val_data):
114 """
115 @brief new version of get_batch that allows for model to be loaded to gpus
116
117 The previous version of get_batch in the Tensor_Prac doesn't have the ability
118 for the GPU to actualy be used, even if the device is present. So this new
119 version will take advantage of that hardware.
120
121 @param split: string either 'train' or 'val' which is used to pick data to batch with
122 @param train_data: the data that is split for training
123 @param val_data: the data that is split for validation
124 @return Tuple(Tensor of inputs, Tensor or targets)
125 """
126 # generate a small batch of data of inputs x and targets y
127 data = train_data if split == "train" else val_data
128 ix = torch.randint(len(data) - block_size, (batch_size,))
129 x = torch.stack([data[i : i + block_size] for i in ix])
130 y = torch.stack([data[i + 1 : i + block_size + 1] for i in ix])
131 # IMPORTANT: This is the part that is needed for the GPU
132 x, y = x.to(device), y.to(device)
133 return x, y

153 def __init__(self, vocab_size):
154 """
155 @brief constructor for BigramLanguageModel that calls super, and then
156 creates a new embedding table of vocab_size by vocab_size
157
158 @param self: the current BigramLanguageModel object
159 @param vocab_size: the length of the vocab_size
160 @return new BigramLanguageModel
161 """
162 super().__init__()
163 # each token directly reads off the logits for hte next token from a tookup table
164 self.token_embedding_table = nn.Embedding(vocab_size, vocab_size)
165
166 def forward(self, idx, targets=None):
167 """
168 @brief goes "ahead" by getting our next set of logits and then returns
169 the logits and loss
170
171 @param self: the current BigramLanguageModel object
172 @param idx: the current context batch
173 @param targets: the tokens that are actually next
174 @return logits, loss: Tuple(logits, loss), which are the logits, and
175 the loss under cross_entropy
176 """
177
178 # idx and targets are both (B, T) tensor of integers
179 logits = self.token_embedding_table(idx) # (B, T, C)
180
181 # negative log likelyhood loss (cross-entropy) how close are logits
182 # to target
183 if targets == None:
184 loss = None
185 else:
186 B, T, C = logits.shape
187 logits = logits.view(B * T, C)
188 targets = targets.view(B * T)
189 loss = F.cross_entropy(logits, targets)
190
191 return logits, loss
192
193 def generate(self, idx, max_new_tokens):
194 """
195 @brief this is the generator of the bigram model, where the model creates tokens
196
197 Idx is the current context of characters in the current batch. And we
198 expent all the B dimension in the T dimension. What is predicted along
199 the T dimension.
200 We take the current idx, focus on the last element in the T dimension.
201 These are then converted to probabilities using softmax, and the sampled
202 using multinomial. We then take the sampled integers, and concat them
203
204 @param self: the current BigramLanguageModel object
205 @param idx: the current context batch (which is dim(B by T))
206 @param max_new_tokens: the amount of new tokens we want to generate
207 @return idx: modified idx with the new generated tokens appended
208 """
209 # idx is (B, T) array of incixes in the current context
210 for _ in range(max_new_tokens):
211 # get the predictions
212 logits, loss = self(idx)
213 # focus only on the last time step
214 logits = logits[:, -1, :] # now dim(B by C)
215 # apply the softmax to get probabilities
216 probs = F.softmax(logits, dim=-1) # dim(B by C)
217 # sample from the distribution
218 idx_next = torch.multinomial(probs, num_samples=1) # (B, 1)
219 # append sampled index to the running sequence
220 idx = torch.cat((idx, idx_next), dim=1) # dim(B by T-1)
221 return idx
