99class BigramLanguageModel(nn.Module):
100
101 def __init__(self, vocab_size):
102 """
103 @brief constructor for BigramLanguageModel that calls super, and then
104 creates a new embedding table of vocab_size by vocab_size
105
106 @param self: the current BigramLanguageModel object
107 @param vocab_size: the length of the vocab_size
108 @return new BigramLanguageModel
109 """
110 super().__init__()
111 # each token directly reads off the logits for hte next token from a tookup table
112 self.token_embedding_table = nn.Embedding(vocab_size, vocab_size)
113
114 def forward(self, idx, targets=None):
115 """
116 @brief goes "ahead" by getting our next set of logits and then returns
117 the logits and loss
118
119 @param self: the current BigramLanguageModel object
120 @param idx: the current context batch
121 @param targets: the tokens that are actually next
122 @return logits, loss: Tuple(logits, loss), which are the logits, and
123 the loss under cross_entropy
124 """
125
126 # idx and targets are both (B, T) tensor of integers
127 logits = self.token_embedding_table(idx) # (B, T, C)
128
129 # negative log likelyhood loss (cross-entropy) how close are logits
130 # to target
131 if targets == None:
132 loss = None
133 else:
134 B, T, C = logits.shape
135 logits = logits.view(B * T, C)
136 targets = targets.view(B * T)
137 loss = F.cross_entropy(logits, targets)
138
139 return logits, loss
140
141 def generate(self, idx, max_new_tokens):
142 """
143 @brief this is the generator of the bigram model, where the model creates tokens
144
145 Idx is the current context of characters in the current batch. And we
146 expent all the B dimension in the T dimension. What is predicted along
147 the T dimension.
148 We take the current idx, focus on the last element in the T dimension.
149 These are then converted to probabilities using softmax, and the sampled
150 using multinomial. We then take the sampled integers, and concat them
151
152 @param self: the current BigramLanguageModel object
153 @param idx: the current context batch (which is dim(B by T))
154 @param max_new_tokens: the amount of new tokens we want to generate
155 @return idx: modified idx with the new generated tokens appended
156 """
157 # idx is (B, T) array of incixes in the current context
158 for _ in range(max_new_tokens):
159 # get the predictions
160 logits, loss = self(idx)
161 # focus only on the last time step
162 logits = logits[:, -1, :] # now dim(B by C)
163 # apply the softmax to get probabilities
164 probs = F.softmax(logits, dim=-1) # dim(B by C)
165 # sample from the distribution
166 idx_next = torch.multinomial(probs, num_samples=1) # (B, 1)
167 # append sampled index to the running sequence
168 idx = torch.cat((idx, idx_next), dim=1) # dim(B by T-1)
169 return idx
